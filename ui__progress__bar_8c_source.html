80static void ui_progress_bar_draw_bar(ui_widget_t* widget) {
81 ui_progress_bar_t* progress_bar = PROGRESS_BAR(widget);
82
83 uint16_t x = widget->render_x;
84 uint16_t y = widget->render_y;
85 uint16_t width = widget->width;
86 uint16_t height = widget->height;
87
88 uint16_t inner_x = x + progress_bar->border_thickness;
89 uint16_t inner_y = y + progress_bar->border_thickness;
90 uint16_t inner_width = width - (2 * progress_bar->border_thickness);
91 uint16_t inner_height = height - (2 * progress_bar->border_thickness);
92
93 uint16_t padding = 4;
94
95 uint32_t bar_color = progress_bar->bar_color;
96
97 // Values normalization:
98 float normalized_ratio = 0.0f;
99 if (progress_bar->max > progress_bar->min) {
100 normalized_ratio = (float)(progress_bar->value - progress_bar->min) / (float)(progress_bar->max - progress_bar->min);
101 }
102
103 // Compute bar movement range:
104 uint16_t min_x = inner_x + padding;
105 uint16_t max_x = inner_x + inner_width - padding;
106
107 // Compute bar position:
108 int16_t bar_x = min_x;
109 uint16_t bar_y = inner_y + padding;
110 int16_t bar_width = (uint16_t)(normalized_ratio * (max_x - min_x));
111 uint16_t bar_height = inner_height - (2 * padding);
112
113 // Draw bar:
114 if (bar_width > 0) {
115 lcd_fill_rect(bar_x, bar_y, bar_width, bar_height, bar_color);
116 }

214bool ui_progress_bar_on_event(ui_widget_t* widget, ui_event_t event) {
215 ui_progress_bar_t* progress_bar = PROGRESS_BAR(widget);
216
217 int16_t new_value = progress_bar->value;
218
219 switch(event) {
220 case UI_EVENT_RIGHT: new_value += progress_bar->step; break;
221 case UI_EVENT_LEFT: new_value -= progress_bar->step; break;
222 default: return false;
223 }
224
225 // Limit to range <min, max>:
226 if (new_value < progress_bar->min) { new_value = progress_bar->min; }
227 if (new_value > progress_bar->max) { new_value = progress_bar->max; }
228
229 progress_bar->value = new_value;
230
231 return true;
232}
