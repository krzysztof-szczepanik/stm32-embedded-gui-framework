38static void ui_toggle_draw_background(ui_widget_t* widget) {
39 ui_toggle_t* toggle = TOGGLE(widget);
40
41 uint16_t x = widget->render_x;
42 uint16_t y = widget->render_y;
43 uint16_t width = widget->width;
44 uint16_t height = widget->height;
45
46 uint16_t inner_x = x + toggle->border_thickness;
47 uint16_t inner_y = y + toggle->border_thickness;
48 uint16_t inner_width = width - (2 * toggle->border_thickness);
49 uint16_t inner_height = height - (2 * toggle->border_thickness);
50
51 // Select color depending on focus state:
52 bool focused = widget->focused;
53
54 uint32_t border_color = focused ? toggle->focus_border_color : toggle->border_color;
55 uint32_t background_color = focused ? toggle->focus_background_color : toggle->background_color;
56
57 // Draw border:
58 if (toggle->border_thickness > 0) {
59 lcd_fill_rect(x, y, width, height, border_color);
60 }
61
62 // Draw background:
63 lcd_fill_rect(x + toggle->border_thickness, y + toggle->border_thickness, width - (2 * toggle->border_thickness), height - (2 * toggle->border_thickness), background_color);
64
65 // Compute separator position:
66 uint16_t separator_x = inner_x + (inner_width / 2) - (toggle->border_thickness / 2);
67
68 // Draw separator:
69 lcd_fill_rect(separator_x, inner_y, toggle->border_thickness, inner_height, border_color);
70}

72static void ui_toggle_draw_label(ui_widget_t* widget) {

85 uint16_t padding = 4;

87 // Compute separator position:
88 uint16_t separator_x = inner_x + (inner_width / 2) - (toggle->border_thickness / 2);
89
90 // Compute left area position and size:
91 uint16_t left_area_x = inner_x;
92 uint16_t left_area_width = separator_x - inner_x;
93
94 // Compute right area position and size:
95 uint16_t right_area_x = separator_x + toggle->border_thickness;
96 uint16_t right_area_width = (inner_x + inner_width) - right_area_x;
97
98 // Select color depending on focus state:
99 bool focused = widget->focused;
100
101 uint32_t background_color = focused ? toggle->focus_background_color : toggle->background_color;
102 uint32_t selected_color = focused ? toggle->focus_selected_color : toggle->selected_color;
103
104 uint32_t left_area_color = (toggle->selected == UI_TOGGLE_LEFT) ? selected_color : background_color;
105 uint32_t right_area_color = (toggle->selected == UI_TOGGLE_RIGHT) ? selected_color : background_color;
106
107 // Draw background for left and right areas:
108 lcd_fill_rect(left_area_x + padding, inner_y + padding, left_area_width - (2 * padding), inner_height - (2 * padding), left_area_color);
109 lcd_fill_rect(right_area_x + padding, inner_y + padding, right_area_width - (2 * padding), inner_height - (2 * padding), right_area_color);
110
111 // Compute label position for left area:
112 uint16_t left_area_text_x = left_area_x + (left_area_width - toggle->left_area_label_font->Width * strlen(toggle->left_area_label)) / 2;
113 uint16_t left_area_text_y = y + (height - toggle->left_area_label_font->Height) / 2 + 2;
114
115 // Compute label position for right area:
116 uint16_t right_area_text_x = right_area_x + (right_area_width - toggle->right_area_label_font->Width * strlen(toggle->right_area_label)) / 2;
117 uint16_t right_area_text_y = y + (height - toggle->right_area_label_font->Height) / 2 + 2;
118
119 // Draw label for left area:
120 lcd_set_font(toggle->left_area_label_font);
121 lcd_set_text_color(toggle->left_area_label_font_color);
122 lcd_set_back_color(left_area_color);
123
124 lcd_display_string_at(left_area_text_x, left_area_text_y, toggle->left_area_label, LCD_ALIGN_LEFT);
125
126 // Draw label for right area:
127 lcd_set_font(toggle->right_area_label_font);
128 lcd_set_text_color(toggle->right_area_label_font_color);
129 lcd_set_back_color(right_area_color);
130
131 lcd_display_string_at(right_area_text_x, right_area_text_y, toggle->right_area_label, LCD_ALIGN_LEFT);
132}
