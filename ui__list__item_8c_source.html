43static bool ui_list_item_focus_in(ui_list_item_t* list_item) {
44
45 // If there is no object -> do nothing:
46 if (list_item->object == NULL) { return false; }
47
48 ui_widget_t* child_widget = list_item->object;
49
50 if (child_widget->focusable) {
51 ui_widget_blur(&list_item->base);
52 ui_widget_focus(child_widget);
53 return true;
54 }
55
56 return false;
57}

75static void ui_list_item_create_object(ui_widget_t* widget) {
76 ui_list_item_t* list_item = LIST_ITEM(widget);
77
78 switch (list_item->object_type) {
79
80 case UI_LIST_ITEM_SLIDER: {
81 ui_slider_t* slider = malloc(sizeof(ui_slider_t));
82 ui_slider_init(slider, 0, 0, 0, 0);
83 list_item->object = &slider->base;
84 break;
85 }
86
87 case UI_LIST_ITEM_SLIDER_TIME: {
88 ui_slider_time_t* slider_time = malloc(sizeof(ui_slider_time_t));
89 ui_slider_time_init(slider_time, 0, 0, 0, 0);
90 list_item->object = &slider_time->base;
91 break;
92 }
93
94 case UI_LIST_ITEM_TOGGLE: {
95 ui_toggle_t* toggle = malloc(sizeof(ui_toggle_t));
96 ui_toggle_init(toggle, 0, 0, 0, 0);
97 list_item->object = &toggle->base;
98 break;
99 }
100
101 case UI_LIST_ITEM_NONE:
102 default:
103 list_item->object = NULL;
104 break;
105 }
106}

108static void ui_list_item_draw_separator(ui_widget_t* widget) {
109 ui_list_item_t* list_item = LIST_ITEM(widget);
110
111 uint16_t x = widget->render_x;
112 uint16_t y = widget->render_y;
113 uint16_t height = widget->height;
114
115 uint16_t inner_x = x + list_item->border_thickness;
116 uint16_t inner_y = y + list_item->border_thickness;
117 uint16_t inner_height = height - (2 * list_item->border_thickness);
118
119 // Select color depending on focus state:
120 bool focused = widget->focused;
121
122 uint32_t separator_color = focused ? list_item->focus_border_color : list_item->border_color;
123
124 // Compute separator position and size:
125 uint16_t separator_x = inner_x + list_item->separator_x_offset;
126 uint16_t separator_y = inner_y;
127 uint16_t separator_width = list_item->border_thickness;
128 uint16_t separator_height = inner_height;
129
130 // Draw separator:
131 if (list_item->border_thickness > 0) {
132 lcd_fill_rect(separator_x, separator_y, separator_width, separator_height, separator_color);
133 }
134}

220static void ui_list_item_draw_right_area_object(ui_widget_t* widget) {
221 ui_list_item_t* list_item = LIST_ITEM(widget);
222
223 uint16_t x = widget->render_x;
224 uint16_t y = widget->render_y;
225 uint16_t width = widget->width;
226 uint16_t height = widget->height;
227
228 uint16_t inner_x = x + list_item->border_thickness;
229 uint16_t inner_y = y + list_item->border_thickness;
230 uint16_t inner_width = width - (2 * list_item->border_thickness);
231 uint16_t inner_height = height - (2 * list_item->border_thickness);
232
233 // Compute right area position and size:
234 uint16_t right_area_x = inner_x + (list_item->separator_x_offset + list_item->border_thickness) + list_item->right_area_padding;
235 uint16_t right_area_y = inner_y + list_item->right_area_padding;
236 uint16_t right_area_width = inner_width - (list_item->separator_x_offset + list_item->border_thickness) - (2 * list_item->right_area_padding);
237 uint16_t right_area_height = inner_height - (2 * list_item->right_area_padding);
238
239 // Draw right area object (slider, toggle, etc.) if it exists:
240 if (list_item->object != NULL) {
241 ui_widget_t* object = list_item->object;
242 object->render_x = right_area_x;
243 object->render_y = right_area_y;
244
245 widget_width(object) = right_area_width;
246 widget_height(object) = right_area_height;
247
248 object->draw(object);
249 }
250}

324bool ui_list_item_on_event(ui_widget_t* widget, ui_event_t event) {
325 ui_list_item_t* list_item = LIST_ITEM(widget);
326
327 // If there is no object -> do nothing:
328 if (list_item->object == NULL) { return false; }
329
330 ui_widget_t* child_widget = list_item->object;
331
332 if (child_widget->focused) {
333 bool event_served = child_widget->on_event(child_widget, event);
334
335 if (event_served) { return true; }
336
337 // If the child did not handle the event -> return focus to list_item:
338 ui_list_item_focus_out(list_item);
339 return true;
340 }
341
342 // If the child widget did not handle the event:
343 switch (event) {
344 case UI_EVENT_SELECT:
345 if (widget->focused) {
346 ui_list_item_focus_in(list_item);
347 return true;
348 }
349 break;
350
351 default:
352 return false;
353 }
354
355 return true;
356}
