189static inline void ui_slider_auto_set_value(ui_slider_t* slider) {
190 int mid = (slider->max + slider->min) / 2;
191 int step = slider->step;
192
193 // rounding down to the nearest step:
194 int aligned = ((mid + (step/2 - 1)) / step) * step;
195 slider->value = aligned;
