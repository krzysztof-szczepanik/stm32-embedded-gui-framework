187static inline void ui_slider_time_auto_set_value(ui_slider_time_t* slider_time) {
188 int mid = (slider_time->max + slider_time->min) / 2;
189 int step = slider_time->step;
190
191 // rounding down to the nearest step:
192 int aligned = ((mid + (step/2 - 1)) / step) * step;
193 slider_time->value = aligned;
194}
