48void lcd_init(void) {
49 // Double initialization required by STM32F429I:
50 BSP_LCD_Init();
51 BSP_LCD_Init();
52
53 uint32_t offset = LCD_WIDTH * LCD_HEIGHT * 4; // calculate foreground offset
54
55 // Clear frame buffers:
56 lcd_clear_buffer(LCD_FRAME_BUFFER);
57 lcd_clear_buffer(LCD_FRAME_BUFFER + offset);
58
59 // Initialize layers:
60 BSP_LCD_LayerDefaultInit(LCD_BACKGROUND_LAYER, LCD_FRAME_BUFFER);
61 BSP_LCD_LayerDefaultInit(LCD_FOREGROUND_LAYER, LCD_FRAME_BUFFER + offset);
62
63 // Set initial colors:
64 BSP_LCD_SetTextColor(LCD_COLOR_BLACK);
65 BSP_LCD_SetBackColor(LCD_COLOR_WHITE);
66
67 // Clear both layers:
68 BSP_LCD_SelectLayer(LCD_FOREGROUND_LAYER);
69 BSP_LCD_Clear(LCD_COLOR_BLACK);
70
71 BSP_LCD_SelectLayer(LCD_BACKGROUND_LAYER);
72 BSP_LCD_Clear(LCD_COLOR_BLACK);
73}
