95
96 // Select color depending on focus state:
97 bool focused = widget->focused;
98
99 uint32_t handle_color = focused ? slider_time->focus_handle_color : slider_time->handle_color;
100
101 // Values normalization:
102 float normalized_ratio = 0.0f;
103 if (slider_time->max > slider_time->min) {
104 normalized_ratio = (float)(slider_time->value - slider_time->min) / (float)(slider_time->max - slider_time->min);
105 }
106
107 // Compute handle movement range:
108 uint16_t min_x = inner_x + padding;
109 uint16_t max_x = inner_x + inner_width - slider_time->handle_width - padding;
110
111 // Compute handle position:
112 int16_t handle_x = min_x + (uint16_t)(normalized_ratio * (max_x - min_x));
113 uint16_t handle_y = inner_y + padding;
114
115 // Draw handle:
116 lcd_fill_rect(handle_x, handle_y, slider_time->handle_width, inner_height - (2 * padding), handle_color);
117}

132 // Update label text based on current slider_time value:

135
136 snprintf(slider_time->label, sizeof(slider_time->label), "%02d:%02d:00", hours, minutes);
137
138 // Compute label position:
139 uint16_t text_x = x + (width - slider_time->label_font->Width * strlen(slider_time->label)) / 2;
140 uint16_t text_y = y + (height - slider_time->label_font->Height) / 2 + 2;
141
142 // Draw label:

144 lcd_set_text_color(slider_time->label_font_color);
145 lcd_set_back_color(background_color);
146
147 lcd_display_string_at(text_x, text_y, slider_time->label, LCD_ALIGN_LEFT);

216bool ui_slider_time_on_event(ui_widget_t* widget, ui_event_t event) {
217 ui_slider_time_t* slider_time = SLIDER_TIME(widget);
218
219 int16_t new_value = slider_time->value;
220
221 switch(event) {
222 case UI_EVENT_RIGHT: new_value += slider_time->step; break;
223 case UI_EVENT_LEFT: new_value -= slider_time->step; break;
224 default: return false;
225 }
226
227 // Limit to range <min, max>:
228 if (new_value < slider_time->min) { new_value = slider_time->min; }
229 if (new_value > slider_time->max) { new_value = slider_time->max; }
230
231 slider_time->value = new_value;
232
233 return true;
