122bool ui_container_focus_next(ui_container_t* container) {
123 // Get currently focused widget:
124 uint8_t start = container->focused_widget_index;
125 ui_widget_t* current_child_widget = container->widgets[start];
126
127 // If this is the last widget -> exit container:
128 if (start == container->widgets_count - 1) {
129 ui_widget_blur(current_child_widget);
130 container->focus_entered = false;
131 return false;
132 }
133
134 // Search for the next focusable widget:
135 for (uint8_t i = start + 1; i < container->widgets_count; i++) {
136 ui_widget_t* child_widget = container->widgets[i];
137
138 if (child_widget->focusable) {
139 ui_widget_blur(current_child_widget);
140 container->focused_widget_index = i;
141 ui_widget_focus(child_widget);
142 return true;
143 }
144 }
145
146 return false;
147}

149bool ui_container_focus_previous(ui_container_t* container) {
150 // Get currently focused widget:

152 ui_widget_t* current_child_widget = container->widgets[start];
153
154 // If this is the first widget -> exit container:
155 if (start == 0) {
156 ui_widget_blur(current_child_widget);

158 return false;

161 // Search for the previous focusable widget:
162 for (int i = start - 1; i >= 0; i--) {

167 container->focused_widget_index = i;

170 }

194void ui_container_update_layout(ui_widget_t* widget) {
195 ui_container_t* container = CONTAINER(widget);
196
197 // If widget has a parent -> global position = parent position + local offset
198 // If no parent -> widget is root and uses its own coordinates
199 if (widget->parent) {
200 widget->render_x = widget->parent->render_x + widget->x;
201 widget->render_y = widget->parent->render_y + widget->y;
202 } else {
203 widget->render_x = widget->x;
204 widget->render_y = widget->y;
205 }
206
207 uint16_t x = widget->render_x;
208 uint16_t y = widget->render_y;
209
210 // Inner area:
211 uint16_t inner_x = x + container->border_thickness + container->padding_left;
212 uint16_t inner_y = y + container->border_thickness + container->padding_top;
213
214 // Update child widgets:
215 for (uint8_t i = 0; i < container->widgets_count; i++) {
216 ui_widget_t* child_widget = container->widgets[i];
217
218 // Local position -> global position:
219 child_widget->render_x = inner_x + child_widget->x;
220 child_widget->render_y = inner_y + child_widget->y;
221
222 // Recursively update layout if child has its own layout function (concerns only containers):
223 if (child_widget->update_layout) {
224 child_widget->update_layout(child_widget);
225 }
226 }
227}

229void ui_container_on_focus(ui_widget_t* widget) {
230 ui_container_t* container = CONTAINER(widget);
231
232 if (container->widgets_count == 0) { return; }
233
234 // We get the last movement direction from the global ui_direction module (this is the direction from which the user "enters" the container):
235 ui_direction_t direction = ui_direction_get();
236 uint8_t* focused_child_widget_index = &container->focused_widget_index;
237
238 switch (direction) {
239 case UI_DIRECTION_DOWN: *focused_child_widget_index = 0; break;
240 case UI_DIRECTION_UP: *focused_child_widget_index = container->widgets_count - 1; break;
241 default: *focused_child_widget_index = 0; break;
242 }
243
244 ui_widget_focus(container->widgets[*focused_child_widget_index]);
245 container->focus_entered = true;
246}

248void ui_container_on_blur(ui_widget_t* widget) {

258bool ui_container_on_event(ui_widget_t* widget, ui_event_t event) {
259 ui_container_t* container = CONTAINER(widget);
260
261 if (container->widgets_count == 0) { return false; }
262
263 ui_widget_t* focused_child_widget = container->widgets[container->focused_widget_index];
264
265 bool event_served = focused_child_widget->on_event(focused_child_widget, event);
266
267 if (event_served) { return true; }
268
269 // If the child did not handle the event, the container tries to move focus:
270 switch(event) {
271 case UI_EVENT_DOWN: return ui_container_focus_next(container);
272 case UI_EVENT_UP: return ui_container_focus_previous(container);
273 default: return false;
274 }
275}
