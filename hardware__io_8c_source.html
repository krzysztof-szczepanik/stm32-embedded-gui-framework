26 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOCEN; // GPIOC clock
27 RCC->APB2ENR |= RCC_APB2ENR_ADC1EN; // ADC1 clock

29 // Configure PC1 and PC2 as analog pins:
30 GPIOC->MODER |= (3U << (1 * 2)) | (3U << (2 * 2)); // set as GPIO_ANALOG
31 GPIOC->PUPDR &= ~((3U << (1 * 2)) | (3U << (2 * 2))); // disable pull-up/down
32
