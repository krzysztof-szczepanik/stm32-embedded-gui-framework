39static void hbox_apply_stretch(ui_hbox_t* hbox, uint16_t inner_x, uint16_t inner_y, uint16_t inner_width, uint16_t inner_height) {
40 ui_container_t* container = &hbox->base;
41
42 uint16_t available_width = inner_width - hbox->spacing * (container->widgets_count - 1);
43 uint16_t per_child_width = available_width / container->widgets_count;
44 uint16_t per_child_height = inner_height;
45
46 uint16_t offset_x = 0;
47
48 for (uint8_t i = 0; i < container->widgets_count; i++) {
49 ui_widget_t* child_widget = container->widgets[i];
50
51 // Position (always applied):
52 child_widget->render_x = inner_x + offset_x;
53 child_widget->render_y = inner_y;
54
55 // Stretch (depends on mode):
56 switch (hbox->stretch) {
57 case UI_STRETCH_NONE:
58 break;
59
60 case UI_STRETCH_WIDTH:
61 child_widget->width = per_child_width;
62 break;
63
64 case UI_STRETCH_HEIGHT:
65 child_widget->height = per_child_height;
66 break;
67
68 case UI_STRETCH_BOTH:
69 child_widget->width = per_child_width;
70 child_widget->height = per_child_height;
71 break;
72 }
73
74 // Recursively update layout if child has its own layout function (concerns only containers):
75 if (child_widget->update_layout) {
76 child_widget->update_layout(child_widget);
77 }
78
79 offset_x += child_widget->width + hbox->spacing;
80 }
81}

83static void hbox_apply_align(ui_hbox_t* hbox, uint16_t inner_x, uint16_t inner_y, uint16_t inner_width, uint16_t inner_height) {
84 ui_container_t* container = &hbox->base;
85
86 // Check if alignment is allowed:
87 bool allow_align_x = (hbox->stretch == UI_STRETCH_NONE || hbox->stretch == UI_STRETCH_HEIGHT);
88 bool allow_align_y = (hbox->stretch == UI_STRETCH_NONE || hbox->stretch == UI_STRETCH_WIDTH);
89
90 uint16_t offset_x = 0;
91
92 // Horizontal alignment:
93 if (allow_align_x) {
94
95 // Calculate total width of child widgets:
96 uint16_t total_child_widgets_width = 0;
97 for (uint8_t i = 0; i < container->widgets_count; i++) {
98 ui_widget_t* child_widget = container->widgets[i];
99 total_child_widgets_width += child_widget->width;
100
101 if (i < container->widgets_count - 1) {
102 total_child_widgets_width += hbox->spacing;
103 }
104 }
105
106 // Calculate offset_x:
107 switch (hbox->align_horizontal) {
108 case UI_ALIGN_CENTER_X: offset_x = (inner_width - total_child_widgets_width) / 2; break;
109 case UI_ALIGN_RIGHT: offset_x = inner_width - total_child_widgets_width; break;
110 case UI_ALIGN_LEFT: break;
111 default: break;
112 }
113 }
114
115 // Horizontal + vertical layout:
116 uint16_t child_widget_x = offset_x;
117 for (uint8_t i = 0; i < container->widgets_count; i++) {
118 ui_widget_t* child_widget = container->widgets[i];
119
120 // Position x (always applied):
121 child_widget->render_x = inner_x + child_widget_x;
122
123 // Vertical alignment:
124 if (allow_align_y) {
125 uint16_t offset_y = 0;
126
127 // Calculate offset_y:
128 switch (hbox->align_vertical) {
129 case UI_ALIGN_CENTER_Y: offset_y = (inner_height - child_widget->height) / 2; break;
130 case UI_ALIGN_BOTTOM: offset_y = inner_height - child_widget->height; break;
131 case UI_ALIGN_TOP: break;
132 default: break;
133 }
134
135 child_widget->render_y = inner_y + offset_y;
136 } else {
137 child_widget->render_y = inner_y;
138 }
139
140 // Recursively update layout if child has its own layout function (concerns only containers):
141 if (child_widget->update_layout) {
142 child_widget->update_layout(child_widget);
143 }
144
145 child_widget_x += child_widget->width + hbox->spacing;
146 }
147}

196bool ui_hbox_focus_next(ui_hbox_t* hbox) {
197 ui_container_t* container = &hbox->base;
198
199 // Get currently focused widget:
200 uint8_t start = container->focused_widget_index;
201 ui_widget_t* current_child_widget = container->widgets[start];
202
203
204 // Search for the next focusable widget:
205 for (uint8_t i = start + 1; i < container->widgets_count; i++) {
206 ui_widget_t* child_widget = container->widgets[i];
207
208 if (child_widget->focusable) {
209 ui_widget_blur(current_child_widget);
210 container->focused_widget_index = i;
211 ui_widget_focus(child_widget);
212 return true;
213 }
214 }
215
216 // If not found - wrap around to the beginning:
217 for (uint8_t i = 0; i < start; i++) {
218 ui_widget_t* child_widget = container->widgets[i];
219
220 if (child_widget->focusable) {
221 ui_widget_blur(current_child_widget);
222 container->focused_widget_index = i;
223 ui_widget_focus(child_widget);
224 return true;
225 }
226 }
227
228 return false;
229}

284void ui_hbox_update_layout(ui_widget_t* widget) {
285 ui_hbox_t* hbox = HBOX(widget);
286 ui_container_t* container = &hbox->base;
287
288 if (container->widgets_count == 0) { return; }
289
290 uint16_t x = widget->render_x;
291 uint16_t y = widget->render_y;
292 uint16_t width = widget->width;
293 uint16_t height = widget->height;
294
295 // Inner area:
296 uint16_t inner_x = x + container->border_thickness + container->padding_left;
297 uint16_t inner_y = y + container->border_thickness + container->padding_top;
298 uint16_t inner_width = width - (2 * container->border_thickness) - container->padding_left - container->padding_right;
299 uint16_t inner_height = height - (2 * container->border_thickness) - container->padding_top - container->padding_bottom;
300
301 if (hbox->stretch) {
302 hbox_apply_stretch(hbox, inner_x, inner_y, inner_width, inner_height);
303 }
304
305 hbox_apply_align(hbox, inner_x, inner_y, inner_width, inner_height);
306}

308void ui_hbox_on_focus(ui_widget_t* widget) {
309 ui_hbox_t* hbox = HBOX(widget);
310 ui_container_t* container = &hbox->base;
311
312 if (container->widgets_count == 0) { return; }
313
314 // We get the last movement direction from the global ui_direction module (this is the direction from which the user "enters" the container):
315 ui_direction_t direction = ui_direction_get();
316 uint8_t* focused_child_widget_index = &container->focused_widget_index;
317
318 switch (direction) {
319 default: *focused_child_widget_index = 0; break;
320 }
321
322 ui_widget_focus(container->widgets[*focused_child_widget_index]);
323 container->focus_entered = true;
324}

337bool ui_hbox_on_event(ui_widget_t* widget, ui_event_t event) {
338 ui_hbox_t* hbox = HBOX(widget);
339 ui_container_t* container = &hbox->base;
340
341 if (container->widgets_count == 0) { return false; }
342
343 ui_widget_t* focused_child_widget = container->widgets[container->focused_widget_index];
344
345 bool event_served = focused_child_widget->on_event(focused_child_widget, event);
346
347 if (event_served) { return true; }
348
349 // If the child did not handle the event, the container tries to move focus:
350 switch(event) {
351 case UI_EVENT_RIGHT: return ui_hbox_focus_next(hbox);
352 case UI_EVENT_LEFT: return ui_hbox_focus_previous(hbox);
353 default: return false;
354 }
