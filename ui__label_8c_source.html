79static void ui_label_draw_label(ui_widget_t* widget) {
80 ui_label_t* label = LABEL(widget);
81
82 uint16_t x = widget->render_x;
83 uint16_t y = widget->render_y;
84 uint16_t width = widget->width;
85 uint16_t height = widget->height;
86
87 uint16_t inner_x = x + label->border_thickness + label->padding_left;
88 uint16_t inner_y = y + label->border_thickness + label->padding_top;
89 uint16_t inner_width = width - (2 * label->border_thickness) - label->padding_left - label->padding_right;
90 uint16_t inner_height = height - (2 * label->border_thickness) - label->padding_top - label->padding_bottom;
91
92 // Select color depending on focus state:
93 bool focused = widget->focused;
94
95 uint32_t background_color = focused ? label->focus_background_color : label->background_color;
96
97 // Select text source:
98 const char* text = label->use_text_buffer ? label->text_buffer : label->text;
99
100 // Compute text size:
101 uint16_t text_width = label->text_font->Width * strlen(text);
102 uint16_t text_height = label->text_font->Height;
103
104 uint16_t text_x = inner_x;
105 uint16_t text_y = inner_y;
106
107 // Horizontal alignment:
108 switch (label->align_horizontal) {
109 case UI_ALIGN_LEFT: text_x = inner_x; break;
110 case UI_ALIGN_CENTER_X: text_x = inner_x + (inner_width - text_width) / 2 + 2; break;
111 case UI_ALIGN_RIGHT: text_x = inner_x + (inner_width - text_width); break;
112 }
113
114 // Vertical alignment:
115 switch (label->align_vertical) {
116 case UI_ALIGN_TOP: text_y = inner_y; break;
117 case UI_ALIGN_CENTER_Y: text_y = inner_y + (inner_height - text_height) / 2 + 2; break;
118 case UI_ALIGN_BOTTOM: text_y = inner_y + (inner_height - text_height); break;
119 }
120
121 // Draw text:
122 lcd_set_font(label->text_font);
123 lcd_set_text_color(label->text_font_color);
124 lcd_set_back_color(background_color);
125
126 lcd_display_string_at(text_x, text_y, text, LCD_ALIGN_LEFT);
127}
