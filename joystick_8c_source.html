57joystick_event_t joystick_get_event(void) {
58
59 // Pressed:
60 if (previous_sw == 1 && joystick_sw == 0) { return JOYSTICK_CLICK_EDGE; } // impulse
61 if (joystick_sw == 0) { return JOYSTICK_CLICK; } // state
62
63 // Up / down:
64 if (joystick_vry < JOYSTICK_THRESHOLD_LOW) { return JOYSTICK_UP; }
65 if (joystick_vry > JOYSTICK_THRESHOLD_HIGH) { return JOYSTICK_DOWN; }
66
67 // Left / right:
68 if (joystick_vrx < JOYSTICK_THRESHOLD_LOW) { return JOYSTICK_LEFT; }
69 if (joystick_vrx > JOYSTICK_THRESHOLD_HIGH) { return JOYSTICK_RIGHT; }
70
71 return JOYSTICK_NONE;
72}
