332void ui_main_screen_update_time() {
333 ui_main_screen_t* main_screen = g_main_screen;
334
335
336 // ================ TIME LABEL =================
337
338 ui_label_value_t* label_value_time = &main_screen->label_value_time;
339 ui_label_t* label_value_time_right_label = &label_value_right_label(label_value_time);
340
341 int32_t time_remaining = screens_data().list_item_time_value;
342 int32_t time_total = screens_data().list_item_time_value_old;
343
344 // Aktualizacja tekstu czasu:
345 label_text_format(label_value_time_right_label, "%02d:%02d:%02d", time_remaining / 3600, (time_remaining % 3600) / 60, time_remaining % 60);
346
347
348 // =============== PROGRESS_BAR ================
349
350 ui_progress_bar_t* progress_bar = &main_screen->progress_bar;
351
352 if (time_total > 0) {
353
354 int32_t elapsed = time_total - time_remaining; // ile minęło
355
356 float ratio = (float)elapsed / (float)time_total; // 0.0 - 1.0
357 int32_t progress = (int32_t)(ratio * 100.0f); // zamiana na 0-100%
358
359 if (progress < 0) { progress = 0; }
360 if (progress > 100) { progress = 100; }
361
362 progress_bar_set_value(progress_bar, progress);
363 }
364
365
366 // ============== TIME END CHECK ===============
367
368 if (time_remaining <= 0 && screens_data().timer_state == TIMER_RUNNING) {
369 ui_main_screen_timer_finished();
370 }
371
372 ui_screen_manager_get_current_screen()->base.invalidated = true;
373}
