70void lcd_swap_buffers(void) {
71
72 // Waiting for the end of the frame (VSYNC) - triggering the interrupt:
73 HAL_LTDC_ProgramLineEvent(&hltdc, LCD_HEIGHT + 3 + 2);
74
75 // CPU waits until the ready_to_swap flag is set to 1, when it is set, it performs the layer swap:
76 while (!ready_to_swap);
77 ready_to_swap = 0;
78
79 if (current_display_layer == LCD_BACKGROUND_LAYER) {
80 // Background was visible -> show foreground:
81 BSP_LCD_SetLayerVisible(LCD_BACKGROUND_LAYER, DISABLE);
82 BSP_LCD_SetLayerVisible(LCD_FOREGROUND_LAYER, ENABLE);
83
84 current_display_layer = LCD_FOREGROUND_LAYER;
85 current_draw_layer = LCD_BACKGROUND_LAYER;
86 } else {
87 // Foreground was visible -> show background:
88 BSP_LCD_SetLayerVisible(LCD_FOREGROUND_LAYER, DISABLE);
89 BSP_LCD_SetLayerVisible(LCD_BACKGROUND_LAYER, ENABLE);
90
91 current_display_layer = LCD_BACKGROUND_LAYER;
92 current_draw_layer = LCD_FOREGROUND_LAYER;
93 }
94
95 // Select the invisible layer for drawing:
96 BSP_LCD_SelectLayer(current_draw_layer);
97
98 // Immediate application of LTDC configuration changes:
99 HAL_LTDC_Reload(&hltdc, LTDC_RELOAD_VERTICAL_BLANKING);
100 while (hltdc.Instance->SRCR != 0UL); // waiting until the reload is complete
101}
