40static void vbox_apply_stretch(ui_vbox_t* vbox, uint16_t inner_x, uint16_t inner_y, uint16_t inner_width, uint16_t inner_height) {
41 ui_container_t* container = &vbox->base;
42
43 uint16_t available_height = inner_height - vbox->spacing * (container->widgets_count - 1);
44 uint16_t per_child_width = inner_width;
45 uint16_t per_child_height = available_height / container->widgets_count;
46
47 uint16_t offset_y = 0;
48
49 for (uint8_t i = 0; i < container->widgets_count; i++) {
50 ui_widget_t* child_widget = container->widgets[i];
51
52 // Position (always applied):
53 child_widget->render_x = inner_x;
54 child_widget->render_y = inner_y + offset_y;
55
56 // Stretch (depends on mode):
57 switch (vbox->stretch) {
58 case UI_STRETCH_NONE:
59 break;
60
61 case UI_STRETCH_WIDTH:
62 child_widget->width = per_child_width;
63 break;
64
65 case UI_STRETCH_HEIGHT:
66 child_widget->height = per_child_height;
67 break;
68
69 case UI_STRETCH_BOTH:
70 child_widget->width = per_child_width;
71 child_widget->height = per_child_height;
72 break;
73 }
74
75 // Recursively update layout if child has its own layout function (concerns only containers):
76 if (child_widget->update_layout) {
77 child_widget->update_layout(child_widget);
78 }
79
80 offset_y += child_widget->height + vbox->spacing;
81 }
82}

84static void vbox_apply_align(ui_vbox_t* vbox, uint16_t inner_x, uint16_t inner_y, uint16_t inner_width, uint16_t inner_height) {
85 ui_container_t* container = &vbox->base;
86
87 // Check if alignment is allowed:
88 bool allow_align_x = (vbox->stretch == UI_STRETCH_NONE || vbox->stretch == UI_STRETCH_HEIGHT);
89 bool allow_align_y = (vbox->stretch == UI_STRETCH_NONE || vbox->stretch == UI_STRETCH_WIDTH);
90
91 uint16_t offset_y = 0;
92
93 // Vertical alignment:
94 if (allow_align_y) {
95
96 // Calculate total height of child widgets:
97 uint16_t total_child_widgets_height = 0;
98 for (uint8_t i = 0; i < container->widgets_count; i++) {
99 ui_widget_t* child_widget = container->widgets[i];
100 total_child_widgets_height += child_widget->height;
101
102 if (i < container->widgets_count - 1) {
103 total_child_widgets_height += vbox->spacing;
104 }
105 }
106
107 // Calculate offset_y:
108 switch (vbox->align_vertical) {
109 case UI_ALIGN_CENTER_Y: offset_y = (inner_height - total_child_widgets_height) / 2; break;
110 case UI_ALIGN_BOTTOM: offset_y = inner_height - total_child_widgets_height; break;
111 case UI_ALIGN_TOP: break;
112 default: break;
113 }
114 }
115
116 // Vertical + horizontal layout:
117 uint16_t child_widget_y = offset_y;
118 for (uint8_t i = 0; i < container->widgets_count; i++) {
119 ui_widget_t* child_widget = container->widgets[i];
120
121 // Position y (always applied):
122 child_widget->render_y = inner_y + child_widget_y;
123
124 // Horizontal alignment:
125 if (allow_align_x) {
126 uint16_t offset_x = 0;
127
128 // Calculate offset_x:
129 switch (vbox->align_horizontal) {
130 case UI_ALIGN_CENTER_X: offset_x = (inner_width - child_widget->width) / 2; break;
131 case UI_ALIGN_RIGHT: offset_x = inner_width - child_widget->width; break;
132 case UI_ALIGN_LEFT: break;
133 default: break;
134 }
135
136 child_widget->render_x = inner_x + offset_x;
137 } else {
138 child_widget->render_x = inner_x;
139 }
140
141 // Recursively update layout if child has its own layout function (concerns only containers):
142 if (child_widget->update_layout) {
143 child_widget->update_layout(child_widget);
144 }
145
146 child_widget_y += child_widget->height + vbox->spacing;
147 }

224void ui_vbox_update_layout(ui_widget_t* widget) {
225 ui_vbox_t* vbox = VBOX(widget);
226 ui_container_t* container = &vbox->base;
227
228 if (container->widgets_count == 0) { return; }
229
230 uint16_t x = widget->render_x;
231 uint16_t y = widget->render_y;
232 uint16_t width = widget->width;
233 uint16_t height = widget->height;
234
235 // Inner area:
236 uint16_t inner_x = x + container->border_thickness + container->padding_left;
237 uint16_t inner_y = y + container->border_thickness + container->padding_top;
238 uint16_t inner_width = width - (2 * container->border_thickness) - container->padding_left - container->padding_right;
239 uint16_t inner_height = height - (2 * container->border_thickness) - container->padding_top - container->padding_bottom;
240
241 if (vbox->stretch) {
242 vbox_apply_stretch(vbox, inner_x, inner_y, inner_width, inner_height);
243 }
244
245 vbox_apply_align(vbox, inner_x, inner_y, inner_width, inner_height);
246}
