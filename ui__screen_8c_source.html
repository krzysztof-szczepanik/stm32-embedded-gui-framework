192void ui_screen_handle_event(ui_screen_t* screen, ui_event_t event) {
193 ui_widget_t* focused_child_widget = screen->widgets[screen->focused_widget_index];
194
195 bool event_served = focused_child_widget->on_event(focused_child_widget, event);
196
197 if (!event_served) {
198 switch(event) {
199 case UI_EVENT_UP:
200 ui_direction_set(UI_DIRECTION_UP);
201 ui_screen_focus_previous(screen);
202
203 // If the screen has only one widget -> refocus:
204 if (screen->widgets_count == 1) {
205 ui_widget_t* child_widget = screen->widgets[0];
206 ui_widget_blur(child_widget);
207 ui_widget_focus(child_widget);
208 }
209
210 break;
211
212 case UI_EVENT_DOWN:
213 ui_direction_set(UI_DIRECTION_DOWN);
214 ui_screen_focus_next(screen);
215
216 // If the screen has only one widget -> refocus:
217 if (screen->widgets_count == 1) {
218 ui_widget_t* child_widget = screen->widgets[0];
219 ui_widget_blur(child_widget);
220 ui_widget_focus(child_widget);
221 }
222
223 break;
224
225 default:
226 break;
227 }

235void ui_screen_draw(ui_widget_t* widget) {
236 ui_screen_t* screen = SCREEN(widget);
237
238 // If nothing requires redrawing -> exit:
239 if (!widget->invalidated) {
240 return;

245
246 // Draw child widgets:
247 for (uint8_t i = 0; i < screen->widgets_count; i++) {
248 ui_widget_t* child_widget = screen->widgets[i];
249 child_widget->draw(child_widget);
250 child_widget->invalidated = false;
251 }
252
253 // Reset invalidation flag:
